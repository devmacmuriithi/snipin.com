                // Life Insights cluster

                { id: 14, title: "Learning Unlearning", content: "Sometimes progress means unlearning what we thought we knew.", tags: ["learning", "growth", "humility"], cluster: 4, size: 7, connections: 3 },
                { id: 15, title: "Present Moments", content: "The present moment is the only time we can actually influence anything.", tags: ["presence", "mindfulness", "action"], cluster: 4, size: 6, connections: 5 }

        // Generate connections based on shared tags and thematic similarity
        function generateConnections(nodes) {
            const connections = [];
            for (let i = 0; i < nodes.length; i++) {
                for (let j = i + 1; j < nodes.length; j++) {
                    const node1 = nodes[i];
                    const node2 = nodes[j];
                    
                    // Calculate similarity based on shared tags
                    const sharedTags = node1.tags.filter(tag => node2.tags.includes(tag));
                    const similarity = sharedTags.length / Math.max(node1.tags.length, node2.tags.length);
                    
                    // Create connection if similarity threshold is met or nodes are in same cluster
                    if (similarity > 0.2 || (node1.cluster === node2.cluster && Math.random() > 0.3)) {
                        connections.push({
                            source: node1.id,
                            target: node2.id,
                            strength: similarity + (node1.cluster === node2.cluster ? 0.3 : 0),
                            sharedTags: sharedTags
                        });
                    }
                }
            }
            return connections;
        }

        sampleData.links = generateConnections(sampleData.nodes);

        // Cluster colors
        const clusterColors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#f9ca24', '#6c5ce7'];
        const clusterNames = ['Personal Growth', 'Technology', 'Philosophy', 'Creativity', 'Life Insights'];

        // Initialize visualization
        const width = window.innerWidth;
        const height = window.innerHeight;

        const svg = d3.select('#visualization')
            .append('svg')
            .attr('width', width)
            .attr('height', height);

        const g = svg.append('g');

        // Add zoom and pan behavior
        const zoom = d3.zoom()
            .scaleExtent([0.3, 3])
            .on('zoom', (event) => {
                g.attr('transform', event.transform);
            });

        svg.call(zoom);

        // Create force simulation
        const simulation = d3.forceSimulation(sampleData.nodes)
            .force('link', d3.forceLink(sampleData.links).id(d => d.id).distance(d => 100 - d.strength * 50))
            .force('charge', d3.forceManyBody().strength(-300))
            .force('center', d3.forceCenter(width / 2, height / 2))
            .force('collision', d3.forceCollide().radius(d => d.size * 3));

        // Create links
        const link = g.append('g')
            .selectAll('line')
            .data(sampleData.links)
            .enter().append('line')
            .attr('class', 'link')
            .attr('stroke-width', d => d.strength * 3);

        // Create nodes
        const node = g.append('g')
            .selectAll('g')
            .data(sampleData.nodes)
            .enter().append('g')
            .attr('class', 'node')
            .call(d3.drag()
                .on('start', dragstarted)
                .on('drag', dragged)
                .on('end', dragended));

        node.append('circle')
            .attr('r', d => d.size * 2)
            .attr('fill', d => clusterColors[d.cluster])
            .attr('stroke', '#ffffff')
            .attr('stroke-width', 2);

        node.append('text')
            .text(d => d.title)
            .attr('text-anchor', 'middle')
            .attr('dy', '.35em')
            .style('font-size', '10px')
            .style('font-weight', 'bold')
            .style('fill', '#ffffff')
            .style('pointer-events', 'none');

        // Tooltip functionality
        const tooltip = d3.select('#tooltip');

        node.on('mouseover', (event, d) => {
            tooltip.style('opacity', 1)
                .style('left', (event.pageX + 10) + 'px')
                .style('top', (event.pageY - 10) + 'px');
            
            document.getElementById('tooltipTitle').textContent = d.title;
            document.getElementById('tooltipContent').textContent = d.content;
            
            const tagsContainer = document.getElementById('tooltipTags');
            tagsContainer.innerHTML = '';
            d.tags.forEach(tag => {
                const tagElement = document.createElement('span');
                tagElement.className = 'tag';
                tagElement.textContent = tag;
                tagsContainer.appendChild(tagElement);
            });
        })
        .on('mouseout', () => {
            tooltip.style('opacity', 0);
        });

        // Click to explore functionality
        let focusedNode = null;

        node.on('click', (event, d) => {
            if (focusedNode === d) {
                resetView();
                return;
            }

            focusedNode = d;
            exploreNode(d);
        });

        function exploreNode(selectedNode) {
            // Find connected nodes
            const connectedNodeIds = new Set();
            connectedNodeIds.add(selectedNode.id);

            sampleData.links.forEach(link => {
                if (link.source.id === selectedNode.id) {
                    connectedNodeIds.add(link.target.id);
                } else if (link.target.id === selectedNode.id) {
                    connectedNodeIds.add(link.source.id);
                }
            });

            // Fade out unconnected nodes and links
            node.transition().duration(500)
                .style('opacity', d => connectedNodeIds.has(d.id) ? 1 : 0.1);

            link.transition().duration(500)
                .style('opacity', d => 
                    (connectedNodeIds.has(d.source.id) && connectedNodeIds.has(d.target.id)) ? 1 : 0.05
                )
                .classed('active', d => 
                    (d.source.id === selectedNode.id || d.target.id === selectedNode.id)
                );

            // Show back button
            document.getElementById('backBtn').classList.add('visible');
        }

        function resetView() {
            focusedNode = null;
            
            node.transition().duration(500)
                .style('opacity', 1);

            link.transition().duration(500)
                .style('opacity', 1)
                .classed('active', false);

            document.getElementById('backBtn').classList.remove('visible');
        }

        // Control buttons
        document.getElementById('clusterBtn').addEventListener('click', () => {
            // Toggle cluster highlighting
            const isActive = document.getElementById('clusterBtn').classList.contains('active');
            if (isActive) {
                node.selectAll('circle').attr('stroke-width', 2);
                document.getElementById('clusterBtn').classList.remove('active');
            } else {
                node.selectAll('circle').attr('stroke-width', 4);
                document.getElementById('clusterBtn').classList.add('active');
            }
        });

        document.getElementById('resonanceBtn').addEventListener('click', () => {
            const isActive = document.getElementById('resonanceBtn').classList.contains('active');
            if (isActive) {
                link.classed('pulse', false);
                document.getElementById('resonanceBtn').classList.remove('active');
            } else {
                link.classed('pulse', true);
                document.getElementById('resonanceBtn').classList.add('active');
            }
        });

        document.getElementById('resetBtn').addEventListener('click', () => {
            resetView();
            svg.transition().duration(750).call(
                zoom.transform,
                d3.zoomIdentity
            );
        });

        document.getElementById('backBtn').addEventListener('click', resetView);

        // Search functionality
        document.getElementById('searchBox').addEventListener('input', (event) => {
            const searchTerm = event.target.value.toLowerCase();
            
            node.style('opacity', d => {
                if (!searchTerm) return 1;
                return (d.title.toLowerCase().includes(searchTerm) || 
                        d.content.toLowerCase().includes(searchTerm) ||
                        d.tags.some(tag => tag.toLowerCase().includes(searchTerm))) ? 1 : 0.2;
            });
        });

        // Update stats
        document.getElementById('nodeCount').textContent = `${sampleData.nodes.length} thoughts`;
        document.getElementById('connectionCount').textContent = `${sampleData.links.length} connections`;

        // Simulation tick
        simulation.on('tick', () => {
            link.attr('x1', d => d.source.x)
                .attr('y1', d => d.source.y)
                .attr('x2', d => d.target.x)
                .attr('y2', d => d.target.y);

            node.attr('transform', d => `translate(${d.x},${d.y})`);
        });

        // Drag functions
        function dragstarted(event, d) {
            if (!event.active) simulation.alphaTarget(0.3).restart();
            d.fx = d.x;
            d.fy = d.y;
        }

        function dragged(event, d) {
            d.fx = event.x;
            d.fy = event.y;
        }

        function dragended(event, d) {
            if (!event.active) simulation.alphaTarget(0);
            d.fx = null;
            d.fy = null;
        }

        // Add some initial animation
        node.style('opacity', 0)
            .transition()
            .duration(1000)
            .delay((d, i) => i * 50)
            .style('opacity', 1);

        link.style('opacity', 0)
            .transition()
            .duration(1000)
            .delay(500)
            .style('opacity', 1);

        // Handle window resize
        window.addEventListener('resize', () => {
            const newWidth = window.innerWidth;
            const newHeight = window.innerHeight;
            
            svg.attr('width', newWidth).attr('height', newHeight);
            simulation.force('center', d3.forceCenter(newWidth / 2, newHeight / 2));
            simulation.alpha(0.3).restart();
        });

        // Add some dynamic behavior - simulate new thoughts appearing
        setTimeout(() => {
            // This could be triggered when user creates new snips
            console.log("üß† SnipNet is ready to visualize your thought resonance network!");
        }, 2000);
